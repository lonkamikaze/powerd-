438  if (this->current) {
439  /* this is not the first call */
440  if (this->argp && this->argp[0] && this->argp[1]) {

442  * of an argument */
443  if (argCount(*this->current) == 0) {
444  /* proceed to the next short option */
445  ++this->argp;

584  args_max = args_max >= len ? args_max : len;

590  auto const sdash = def.sparam ? '-' : ' ';
591  auto const lparam = def.lparam ? def.lparam : "";
592  auto const ldash = lparam[0] ? '-' : ' ';
593  auto const sep = def.sparam && lparam[0] ? ',' : ' ';

597  /* get arguments */
598  auto len = args_max;
599  for (auto it = def.args; it && *it; ++it, --len) {
600  result += *it == ',' ? ' ' : *it;
601  }

622  std::string show(int const i, int const n = 1) const {
623  /* select a whole argument */
624  char const * const select = (*this)[i];
625  /* if the current option (i == 0) is requested, pick
626  * up the pointer to the current short option character */
627  char const * const argp = i == 0 ? this->argp : nullptr;
628  using std::string;
629  string cmd; /* command and arguments string */
630  string ul; /* underlining string */
631  int hilight = 0; /* #args left to underline */
632  bool found = false; /* whether argument i was found */
633  /* build cmd and ul string */
634  for (int p = 0; p < this->argc; ++p) {
635  /* build each argument character wise */
636  for (auto it = this->argv[p];; ++it) {
637  /* underlining character */
638  char ulc = hilight ? '~' : ' ';
639  /* underline short option */
640  if (argp && it == argp) {
641  hilight = n > 0 ? n - 1 : -1;
642  ulc = '^';
643  found = true;
644  }
645  /* underline long option / argument */
646  if (!argp && it == select) {
647  hilight = n > 0 ? n : -1;
648  ulc = '^';
649  found = true;
650  }
651  /* add current character,
652  * continue to stay in the loop */
653  switch (*it) {
654  case 0:
655  /* end of argument,
656  * add a space behind the argument */
657  cmd += ' ';
658  /* underline only if it's the first
659  * character to underline */
660  ul += ulc == '^' ? '^' : ' ';
661  break;
662  case '\t':
663  /* symbolic tab */
664  cmd += "\\t";
665  ul += ulc += ulc;
666  continue;
667  case '\n':
668  /* symbolic newline */
669  cmd += "\\n";
670  ul += ulc += ulc;
671  continue;
672  case ' ':
673  case '\\':
674  /* escape */
675  cmd += '\\';
676  ul += ulc;
677  /*[[fallthrough]];*/
678  default:
679  /* regular character */
680  cmd += *it;
681  ul += ulc;
682  continue;
683  }
684  break;
685  }
686  hilight > 0 && --hilight;
687  }
688  /* the selected argument must be behind the last argument,
689  * e.g. because the last parameter is missing an argument
690  * or the state is OPT_DONE */
691  if (!found) {
692  cmd += ' ';
693  ul += '^';
694  }
