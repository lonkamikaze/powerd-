597  }

634  std::string show(int const i, int const n = 1) const {
635  /* select a whole argument */
636  char const * const select = (*this)[i];
637  /* if the current option (i == 0) is requested, pick
638  * up the pointer to the current short option character */
639  char const * const argp = i == 0 ? this->argp : nullptr;
640  using std::string;
641  string cmd; /* command and arguments string */
642  string ul; /* underlining string */
643  int hilight = 0; /* #args left to underline */
644  bool found = false; /* whether argument i was found */
645  /* build cmd and ul string */
646  for (int p = 0; p < this->argc; ++p) {
647  /* build each argument character wise */
648  for (auto it = this->argv[p];; ++it) {
649  /* underlining character */
650  char ulc = hilight ? '~' : ' ';
651  /* underline short option */
652  if (argp && it == argp) {
653  hilight = n > 0 ? n - 1 : -1;
654  ulc = '^';
655  found = true;
656  }
657  /* underline long option / argument */
658  if (!argp && it == select) {
659  hilight = n > 0 ? n : -1;
660  ulc = '^';
661  found = true;
662  }
663  /* add current character,
664  * continue to stay in the loop */
665  switch (*it) {
666  case 0:
667  /* end of argument,
668  * add a space behind the argument */
669  cmd += ' ';
670  /* underline only if it's the first
671  * character to underline */
672  ul += ulc == '^' ? '^' : ' ';
673  break;
674  case '\t':
675  /* symbolic tab */
676  cmd += "\\t";
677  ul += ulc += ulc;
678  continue;
679  case '\n':
680  /* symbolic newline */
681  cmd += "\\n";
682  ul += ulc += ulc;
683  continue;
684  case ' ':
685  case '\\':
686  /* escape */
687  cmd += '\\';
688  ul += ulc;
689  /*[[fallthrough]];*/
690  default:
691  /* regular character */
692  cmd += *it;
693  ul += ulc;
694  continue;
695  }
696  break;
697  }
698  hilight > 0 && --hilight;
699  }
700  /* the selected argument must be behind the last argument,
701  * e.g. because the last parameter is missing an argument
702  * or the state is OPT_DONE */
703  if (!found) {
704  cmd += ' ';
705  ul += '^';
706  }
707  return cmd + '\n' + ul;
708  }
